Connection Setup 
// WebSocket Connection
const apiUrl = 'wss://api.nebelus.ai';
const agentId = 'your-agent-id';
const apiKey = 'your-api-key';
const threadId = null; // or existing thread ID

let wsUrl;
if (threadId) {
  // Continue existing thread
  wsUrl = `${apiUrl}/ws/agents/${agentId}/threads/${threadId}/?api_key=${apiKey}`;
} else {
  // Create new thread
  wsUrl = `${apiUrl}/ws/agents/${agentId}/chat/?api_key=${apiKey}`;
}

const ws = new WebSocket(wsUrl);

ws.onopen = () => console.log('Connected!');
ws.onerror = (error) => console.error('Error:', error);
ws.onclose = (event) => console.log('Disconnected:', event.code);

Handle Server Events 
// WebSocket event handling
ws.onmessage = (event) => {
  const message = JSON.parse(event.data);

  switch (message.event) {
    case 'connection':
      const threadId = message.data.thread_id;
      console.log('Thread:', threadId);
      break;

    case 'message_start':
      console.log('Assistant is responding...');
      break;

    case 'content_block':
      const { content_type, state, data } = message.data;
      if (content_type === 'text' && state === 'delta') {
        console.log(data.text); // Streaming text
      }
      break;

    case 'message_stop':
      console.log('Response complete');
      break;

    case 'usage_metadata':
      console.log('Tokens:', message.data);
      break;

    case 'error':
      console.error('Error:', message.data.message);
      break;
  }
};

// Cancel ongoing generation
ws.send(JSON.stringify({ type: 'cancel' }));

// Keep-alive ping (every 30 seconds)
setInterval(() => {
  if (ws.readyState === WebSocket.OPEN) {
    ws.send(JSON.stringify({ type: 'ping' }));
  }
}, 30000);
